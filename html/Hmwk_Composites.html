%Define given properties in GPa for moduli
Em = 4.62; Ef1 = 233; Ef2 = 23.1; Gf12 = 8.96; Gf23 = 8.27;
Num = .36; Nuf12 = .2; Nuf23 = .4; Vf = .6; Vm = 1-Vf;
%Do I need to calculate Gm since it's not given? Gm = Em/(2*(1+Num));
%Calculate the requested values using RoM
E1 = Ef1*Vf+Em*Vm;
Nu12 = Num*Vm+Nuf12*Vf;
E2 = (Ef1*Em)/(Vm*Ef1+Vf*Em);
G12 = (Gf12*Gf23)/(Vm*Gf12+Vf*Gf23);

C2

%Using the same values as C1, plot them with varying Vf
Vf = 0:.1:1;
for i=1:length(Vf)
    Vm = 1-Vf(i);
    E1_plot(i) = Ef1*Vf(i)+Em*Vm;
    Nu12_plot(i) = Num*Vm+Nuf12*Vf(i);
    E2_plot(i) = (Ef1*Em)/(Vm*Ef1+Vf(i)*Em);
    G12_plot(i) = (Gf12*Gf23)/(Vm*Gf12+Vf(i)*Gf23);
end
figure(1);plot(Vf,E1_plot);figure(2);plot(Vf,Nu12_plot);
figure(3);plot(Vf,E2_plot);figure(4);plot(Vf,G12_plot);
